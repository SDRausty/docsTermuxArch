startarch c[md] cmd: Run Arch Linux command from Termux as root user.

startarch u[ser]|l[ogin] user: Login as user.  Use startarch addauser user first to create this user and user's home directory.

#!/bin/env bash
# Copyright 2017-2018 by SDRausty. All rights reserved.  🌎 🌍 🌏 🌐 🗺
# Hosting sdrausty.github.io/TermuxArch courtesy https://pages.github.com.
# https://sdrausty.github.io/TermuxArch/README has info about this project.
# https://sdrausty.github.io/TermuxArch/CONTRIBUTORS Thank you for your help.
# Pacman install packages wrapper with system update.
################################################################################
# IFS=$'\n\t'
set -Eeuo pipefail
# shopt -s nullglob globstar
unset LD_PRELOAD
VERSIONID=v1.6.3.id2148

_TRPERR_() {  # Run on script error.
        local RV="$?"
        printf "\e[?25h\n\e[1;48;5;138m %s\e[0m\n\n" "${0##*/} WARNING:  Generated script signal ${RV:-unknown} near or at line number ${1:-unknown} by `${2:-command}`!"
        exit 201
}

_TRPET_() {  #  Run on exit.
        local RV="$?"
        printf "\a\a\a\a"
        sleep 0.4
        if [[ "$RV" = 0 ]] ; then
                printf "\a\e[0;32m%s\e[1;34m: \a\e[1;32m%s\e[0m\n\n\a\e[0m" "${0##*/} $@ v1.6.3.id2148" "DONE 🏁 "
                printf "\e]2; %s: %s \007" "${0##*/} $@" "DONE 🏁 "
        else
                printf "\a\e[0;32m%s \a\e[0m%s\e[1;34m: \a\e[1;32m%s\e[0m\n\n\a\e[0m" "${0##*/} $@ v1.6.3.id2148" "[Exit Signal $RV]" "DONE  🏁 "
                printf "\033]2; %s: %s %s \007" "${0##*/} $@" "[Exit Signal $RV]" "DONE 🏁 "
        fi
        printf "\e[?25h\e[0m"
        set +Eeuo pipefail
        exit
}

_TRPSIG_() {  # Run on signal.
        printf "\e[?25h\e[1;7;38;5;0m${0##*/} WARNING:  Signal $? received!\e[0m\n"
        exit 211
}

_TRPQ_() {  #   Run on quit.
        printf "\e[?25h\e[1;7;38;5;0m${0##*/} WARNING:  Quit signal $? received!\e[0m\n"
        exit 221
}

trap '_TRPERR_ $LINENO $BASH_COMMAND $?' ERR
trap _TRPET_ EXIT
trap _TRPSIG_ HUP INT TERM
trap _TRPQ_ QUIT

declare ARGS="$@"

_TRPET_() { # on exit
printf "\e[?25h\e[0m"
set +Eeuo pipefail
        _PRINTTAIL_ "$ARGS"
}

_PRINTTAIL_() {
printf "\\a\\n\\e[0;32m%s %s %s\\a\\e[1;34m: \\a\\e[1;32m%s\\e[0m 🏁  \\n\\n\\a\\e[0m" "TermuxArch $(basename "$0")" "$ARGS" "$VERSIONID" "DONE"
printf '\033]2;  🔑🗝 TermuxArch '"$(basename "$0") $ARGS"' 📱 \007'
}

trap _TRPET_ EXIT
## pci begin ###################################################################

printf "\\n\\e[1;32m==> \\e[1;37m%s \\e[1;32m%s %s %s \\e[0m%s…\\n\\n" "Running" "TermuxArch $(basename "$0")" "$ARGS" "$VERSIONID"
if [[ -z "${1:-}" ]] ; then
pacman -Syu --noconfirm --color=always
elif [[ "$1" = "e" ]] ; then
pacman -Syu base base-devel emacs "${@:2}" --noconfirm --color=always
elif [[ "$1" = "e8" ]] ; then
pacman -Syu base base-devel emacs jdk8-openjdk "${@:2}" --noconfirm --color=always
elif [[ "$1" = "e10" ]] ; then
pacman -Syu base base-devel emacs jdk10-openjdk "${@:2}" --noconfirm --color=always
else
pacman -Syu "$@" --noconfirm --color=always
fi
